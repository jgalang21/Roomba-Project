22 void ir_init(void){
23  SYSCTL_RCGCGPIO_R |= SYSCTL_RCGCGPIO_R1; //enable ADC 0 module on port D
24  SYSCTL_RCGCADC_R |= 0x1; //enable clock for ADC
25  GPIO_PORTB_AFSEL_R |= 0x01; //enable port D pin 0 to work as alternate functions
26  GPIO_PORTB_DIR_R &= 0b11111110; //set pin to input - 0
27  GPIO_PORTB_DEN_R |= 0x00; //disable PB4 digital function
28  GPIO_PORTB_AMSEL_R |= 0x01; //disable analog isolation for the pin
29  GPIO_PORTB_ADCCTL_R = 0x00; //initialize the port trigger source as processor (default)
30 
31  ADC0_ACTSS_R &= ~ADC_ACTSS_ASEN1; //disable SS1 sample sequencer to configure it
32  ADC0_EMUX_R = ADC_EMUX_EM1_PROCESSOR; //initialize the ADC trigger source as processor (default)
33  ADC0_SSMUX1_R |= 0x000A; //set 1st sample to use the AIN10 ADC pin
34  ADC0_SSCTL0_R |= (ADC_SSCTL1_IE0 | ADC_SSCTL1_END0); //enable raw interrupt status
35  ADC0_SAC_R |= ADC_SAC_AVG_64X; //enable oversampling to average
36  ADC0_ACTSS_R |= ADC_ACTSS_ASEN1; //re-enable ADC0 SS1
37 }

46 int ir_read(void){
47  ADC0_ACTSS_R &= ~ADC_ACTSS_ASEN1; //disable ADC0SS1 sample sequencer to configure it
48  ADC0_SSMUX1_R |= 0; //set 1st sample to use the channel ADC pin
49  ADC0_ACTSS_R |= ADC_ACTSS_ASEN1; //re-enable ADC0 SS1
50  ADC0_PSSI_R=ADC_PSSI_SS1; //initiate SS1 conversion
51 
52  while((ADC_RIS_INR1) == 0){} //wait for ADC conversion to be complete
53  ADC0_ISC_R=ADC_ISC_IN1; //clear interrupt
54 
55  return ADC0_SSFIFO1_R;
56 }
63 float ir_getDistance(void){
64  int i = 0, increments = 100;
65  isr_int_value = 0;
66 
67  //Compute a running average of 100 samples and display that value
68  for(i=0;i<increments;i++){
69  timer_waitMicros(500); //samples every 1 millisecond
70  isr_int_value += ir_read();
71  }
72  int value = isr_int_value/increments; //average of samples
73 
74  //equation from excel y = 105375 x ^(-1.153) where y = distance (cm) and x = quantized value
75  float distance = 105375 * pow(value,-1.153);
76  return distance;
