171 {
172  uint8_t sensorBuffer[SENSOR_PACKET_SIZE];
173 
174  //Query list of sensors
175  oi_uartSendChar(OI_OPCODE_SENSORS);
176  oi_uartSendChar(OI_SENSOR_PACKET_GROUP100);
177 
178  // Read all the sensor data
179  uint8_t i;
180  for (i = 0; i < SENSOR_PACKET_SIZE; i++) {
181  // read each sensor byte
182  sensorBuffer[i] = oi_uartReceive();
183  }
184 
185  //Parse the sensor data into the struct
186  oi_parsePacket(self, sensorBuffer);
187 
188  timer_waitMillis(25); // reduces USART errors that occur when continuously transmitting/receiving min wait time=15ms
189 }

371 void oi_uartInit(void)
372 {
373  //Calculated Baudrate for 115200;
374  uint16_t iBRD = 0x08;//BRD=SYSCLK/((ClkDiv)(BaudRate)), HSE=0 ClkDiv=16, BaudRate=115,200
375  uint16_t fBRD = 44;//Fractional remainder is 0.6805, DIVFRAC = (.6805)(64)+0.5 = 44
376 
377  SYSCTL_RCGCGPIO_R |= SYSCTL_RCGCGPIO_R2; //enable GPIO Port C
378 
379  SYSCTL_RCGCUART_R |= SYSCTL_RCGCUART_R4; //enable UART4
380 
381  GPIO_PORTC_AFSEL_R |= (BIT4 | BIT5); //Enable alternate function on PC6,PC7
382  GPIO_PORTC_PCTL_R |= 0x00110000; //Enable function 1 (UART Rx/Tx) on PC4,PC5
383  GPIO_PORTC_DEN_R |= (BIT4 | BIT5); //Enable PC4,5 data
384  GPIO_PORTC_DIR_R |= BIT5; //Set pin direction to output on PC5
385  GPIO_PORTC_DIR_R &= ~BIT4; //Set pin direction to input on PC4
386 
387  UART4_CTL_R &= ~(UART_CTL_UARTEN); //Disable UART4 while we mess with it
388 
389  UART4_IBRD_R = iBRD;
390  UART4_FBRD_R = fBRD;
391 
392  UART4_LCRH_R = UART_LCRH_WLEN_8; //8 bit, 1 stop, no parity, no FIFO
393  UART4_CC_R = UART_CC_CS_SYSCLK; //Use System Clock
394  UART4_CTL_R = UART_CTL_RXE | UART_CTL_TXE | UART_CTL_UARTEN; //Enable Rx, Tx and UART module
395 }

462  buffer[ptr++] = c;
463 
464  //Do a string compare
465  buffer[ptr] = '\0';
466  if(ptr >= FIRM_STRLEN && !strcmp(buffer + ptr - FIRM_STRLEN, FIRM_STR)) {
467  ptr = 0;
468  //Firmware version incoming
469  while( (c = oi_uartReceive()) != FIRM_END ) {
470  firmware[ptr++] = c;
471  }
472 
473  break;
474  }
475  }
476 
477  firmware[ptr] = '\0';
478 
479  return firmware;

483 void oi_shutoff_init(void){
484  //PF0 user switch - OI shut down
485  SYSCTL_RCGCGPIO_R |= SYSCTL_RCGCGPIO_R5; //enable clock to GPIO F
486  GPIO_PORTF_LOCK_R = 0x4C4F434B;//unlock PF0
487  GPIO_PORTF_CR_R |= 0x01;//allow writes PF0 DEN to be commited
488  GPIO_PORTF_DEN_R |= BIT0; //enable digital on pin 0
489  GPIO_PORTF_DIR_R |= ~BIT0; //set pin 0 to input
490  GPIO_PORTF_IBE_R |= BIT0; //both edges can trigger
491  GPIO_PORTF_IEV_R |= BIT0; //enable rising edge trigger
492  GPIO_PORTF_ICR_R |= BIT0; //clear interrupt if there is already one
493  GPIO_PORTF_IM_R |= BIT0; //unmask interrupt
494 
495  NVIC_EN0_R |= 0x40000000; //enable IRQ by setting bit 30
496  IntRegister(INT_GPIOF, GPIOF_Handler);//tell cpu to use ISR handler for GPIOF
497  IntMasterEnable();//enable global interrupts
498 }

514 int getDegrees(oi_t *self){
515  static int iterations = 0;
516  static int prevLeft = 0;
517  static int prevRight = 0;
518 
519 
520  //if(self->leftEncoderCount == 0 || self->rightEncoderCount == 0){ // update was called with no movement of the bot
521  // return 0;
522  //}
523 
524  if((self->leftEncoderCount == prevLeft) && (self->rightEncoderCount == prevRight)){ // if the bot has not moved since previous update
525  prevLeft = self->leftEncoderCount;
526  prevRight = self->rightEncoderCount;
527  return 0;
528  }
529  //ignore the first run, such that we do not have prevLeft or right=0, this would give a very large degree moved.
530  else if(iterations == 0){
531  prevLeft = self->leftEncoderCount;
532  prevRight = self->rightEncoderCount;
533  iterations++;
534  return 0;
535  }
536 
537  //get distance moved in mm for the left and right wheel
538  // equation: ticks * (1/508)*72pi
539  //update the previous values to be correct
540  int distLeft = (self->leftEncoderCount - prevLeft)*(0.445265);
541  int distRight = (self->rightEncoderCount - prevRight)*(0.445265);
542  prevLeft = self->leftEncoderCount;
543  prevRight = self->rightEncoderCount;
544 
545 
546  //calculate the degree travelled by (right-left)/wheel base in mm
547  double deg = (distRight - distLeft)/178.5;//change 178.5 to improve accuracy of turning if needed
548  deg = (deg*180)/M_PI;
549 
550  return (deg);
551 
552 }
